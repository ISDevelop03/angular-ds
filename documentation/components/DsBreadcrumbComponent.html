import { Component, Input } from '@angular/core';
import { breadcrumb } from './theme';

export interface Page {
  id: string;
  name: string;
  href?: string;
  current?: boolean;
}

/**
 * BreadcrumbComponent
 *
 * Live demo:
 * <example-url>/demo/ds-breadcrumb.component.html</example-url>
 */
@Component({
  selector: 'ds-breadcrumb',
  templateUrl: './breadcrumb.component.html',
})
export class DsBreadcrumbComponent {
  @Input() nextUrl?: string = '';
  @Input() nextClassName?: string = '';
  @Input() prevUrl?: string = '';
  @Input() prevClassName?: string = '';
  @Input() homeUrl?: string = '';
  @Input() className: string = '';
  @Input() variant: keyof typeof breadcrumb = 'default';
  @Input() pages: Page[] = [];
  @Input() showLimit: boolean = false;

  get list(): Page[] {
    if (this.homeUrl && this.pages.length > 2 && this.showLimit) {
      return this.pages.slice(0, 1);
    } else if (this.pages.length > 3 && this.showLimit) {
      return this.pages.slice(0, 2);
    }
    return this.pages;
  }

  get lastPage(): Page {
    return this.pages[this.pages.length - 1];
  }

  get theme() {
    return breadcrumb[this.variant];
  }
}

<div id="ds-breadcrumb" [ngClass]="theme.wrapper" [class]="className">
  <a [href]="prevUrl" [class]="prevClassName">
    <ds-icon id="arrow-left" size="12px"> </ds-icon>
  </a>
  <a [href]="nextUrl" [class]="nextClassName">
    <ds-icon id="arrow-right" size="12px"> </ds-icon>
  </a>

  <ol role="list" [ngClass]="theme.list">
    <li *ngIf="homeUrl" [ngClass]="theme.listElement">
      <a [href]="homeUrl" [ngClass]="theme.link">
        <span [ngClass]="theme.link"> Accueil </span>
      </a>
    </li>

    <ng-container *ngFor="let page of list; let i = index">
      <li [ngClass]="theme.listElement">
        <span *ngIf="i !== 0 || homeUrl" [ngClass]="theme.separator">/</span>

        <a
          [href]="page.href || '#.'"
          [attr.aria-current]="page.current ? 'page' : null"
          [ngClass]="page.current ? theme.linkActive : theme.link"
        >
          {{ page.name }}
        </a>
      </li>
    </ng-container>

    <ng-container *ngIf="list.length !== pages.length">
      <li [ngClass]="theme.listElement">
        <span [ngClass]="theme.separator">/</span>
        <span [ngClass]="theme.link"> ... </span>
      </li>

      <li [ngClass]="theme.listElement">
        <span [ngClass]="theme.separator">/</span>
        <a
          [href]="lastPage.href || '#.'"
          [ngClass]="lastPage.current ? theme.linkActive : theme.link"
          [attr.aria-current]="lastPage.current ? 'page' : null"
        >
          <span [ngClass]="theme.link">
            {{ lastPage.name }}
          </span>
        </a>
      </li>
    </ng-container>
  </ol>
</div>
