import {
  Component,
  EventEmitter,
  Input,
  Output,
  TemplateRef,
} from '@angular/core';
import { theme } from './theme';

export interface ITab {
  title: string;
  panel: string | TemplateRef<any>;
  href?: string;
  disabled?: boolean;
}

/**
 * TabsComponent
 *
 * Live demo:
 * <example-url>/demo/ds-tabs.component.html</example-url>
 */
@Component({
  selector: 'ds-tabs',
  templateUrl: './tabs.component.html',
})
export class DsTabsComponent {
  @Input() items: ITab[] = [];
  @Input() className: string = '';
  @Input() variant: keyof typeof theme = 'default';
  @Input() rightContent?: TemplateRef<any>;

  @Output() onSelect = new EventEmitter<any>();

  @Input() selectedIndex = 0;
  theme = theme;

  selectTab(event: Event, tab: ITab, index: number) {
    if (tab.disabled || tab.href) {
      return;
    }

    this.selectedIndex = index;
    this.onSelect.emit({ event, tab, index });
  }

  isSelected(index: number): boolean {
    return this.selectedIndex === index;
  }

  isTemplate(ref: any): ref is TemplateRef<any> {
    return ref instanceof TemplateRef;
  }
}

<div id="ds-tabs" [ngClass]="[theme[variant].wrapper, className]">
  <div class="flex items-center justify-between">
    <div [ngClass]="[theme[variant].list.container]">
      <ng-container *ngFor="let tab of items; let idx = index">
        <button
          *ngIf="!tab.href; else linkTab"
          (click)="selectTab($event, tab, idx)"
          [ngClass]="[
            theme[variant].list.tab.base,
            isSelected(idx)
              ? theme[variant].list.tab.selected
              : theme[variant].list.tab.unselected,
            tab.disabled ? 'cursor-not-allowed opacity-40' : ''
          ]"
          [attr.aria-selected]="isSelected(idx) ? 'true' : 'false'"
          [attr.aria-disabled]="tab.disabled ? 'true' : null"
          [attr.disabled]="tab.disabled ? true : null"
        >
          {{ tab.title }}
        </button>
  
        <ng-template #linkTab>
          <a
            [href]="tab.href"
            [ngClass]="[
              theme[variant].list.tab.base || '',
              isSelected(idx)
                ? theme[variant].list.tab.selected || ''
                : theme[variant].list.tab.unselected || ''
            ]"
            [attr.aria-disabled]="tab.disabled ? 'true' : null"
            [attr.disabled]="tab.disabled ? true : null"
          >
            {{ tab.title }}
          </a>
        </ng-template>
      </ng-container>
    </div>
    <ng-container *ngIf="rightContent">
      <ng-container *ngTemplateOutlet="rightContent"></ng-container>
    </ng-container>
  </div>

  <div [ngClass]="[theme[variant].panels.container]">
    <div
      *ngFor="let tab of items; let idx = index"
      [hidden]="!isSelected(idx)"
      [ngClass]="theme[variant].panels.panel.base || ''"
    >
      <ng-container *ngIf="isTemplate(tab.panel); else showHtml">
        <ng-container *ngTemplateOutlet="tab.panel"></ng-container>
      </ng-container>
      <ng-template #showHtml>
        <div [innerHTML]="tab.panel"></div>
      </ng-template>
    </div>
  </div>
</div>
