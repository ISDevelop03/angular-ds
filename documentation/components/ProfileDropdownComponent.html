import { Component, Input } from '@angular/core';
import { positions } from './theme';
import { ISelectItem } from './types';

/**
 * Profile-dropdownComponent
 *
 * Live demo:
 * <example-url>/demo/ds-profile-dropdown.component.html</example-url>
 */
@Component({
  selector: 'ds-profile-dropdown',
  templateUrl: './profile-dropdown.component.html',
})
export class ProfileDropdownComponent {
  @Input() className?: string;
  @Input() ulClassName?: string;
  @Input() list: ISelectItem[];
  @Input() children: string;
  @Input() position?: keyof typeof positions;
  @Input() suffix?: string;
  @Input() prefix?: string;

  isOpen: boolean = false;

  handleOpen = () => (this.isOpen = true);
  handleClose = () => (this.isOpen = false);

  //Positions
  positions = positions;
}

<div
  id="ds-profile-dropdown"
  [ngClass]="['relative overflow-visible z-[1100]', className || '']"
>
  <div tabIndex="0" role="button" (click)="handleOpen()">
    <ng-content></ng-content>
  </div>

  <div
    *ngIf="isOpen"
    (click)="handleClose()"
    tabIndex="0"
    role="button"
    class="fixed inset-0 z-[999] isolate bg-black/30"
  ></div>
  <ul
    *ngIf="isOpen"
    [ngClass]="[
      'absolute flex z-[1000] border border-neutral-200 flex-col text-custom-black overflow-hidden shadow-dropdown bg-level-4 rounded-lg w-max',
      positions[position] || '',
      ulClassName || ''
    ]"
  >
    <ng-content select="[prefix]"></ng-content>
    <ng-container *ngFor="let item of list">
      <div
        *ngIf="!item?.href"
        tabIndex="{0}"
        role="button"
        (click)="item?.onClick()"

      >

          [ngClass]="[
            'inline-flex justify-start items-center gap-3 px-4 py-2.5',
            item.className || ''
          ]"
        >

            *ngIf="item?.icon"
            [id]="item?.icon"
            class="size-4"
          ></ds-icon>
          <span class="inline-block text-[14px]/[16px] font-[400]">
            {{ item?.label }}
          </span>
        </li>
      </div>
